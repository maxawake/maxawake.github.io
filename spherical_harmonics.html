import numpy as np
import matplotlib.pyplot as plt

def r(theta, phi):
    Ys = np.sqrt(1/(4*np.pi))
    Ypx = np.sqrt(3/(4*np.pi))*np.sin(theta)*np.cos(phi)
    Ypy = np.sqrt(3/(4*np.pi))*np.sin(theta)*np.sin(phi)
    Ypz = np.sqrt(3/(4*np.pi))*np.cos(theta)

    psi11 = np.sqrt(1/2)*(Ys+Ypx)
    psi12 = np.sqrt(1/2)*(Ys-Ypx)

    psi21 = 1/np.sqrt(3)*Ys+1/np.sqrt(6)*Ypx+1/np.sqrt(2)*Ypy
    psi22 = 1/np.sqrt(3)*Ys+1/np.sqrt(6)*Ypx-1/np.sqrt(2)*Ypy
    psi23 = 1/np.sqrt(3)*Ys-2/np.sqrt(6)*Ypx

    psi31 = 1/2*(Ys+Ypx+Ypy+Ypz)
    psi32 = 1/2*(Ys+Ypx-Ypy-Ypz)
    psi33 = 1/2*(Ys-Ypx+Ypy-Ypz)
    psi34 = 1/2*(Ys-Ypx-Ypy+Ypz)

    psis = np.abs(np.array([psi11,psi12, psi21, psi22, psi23, psi31, psi32, psi33, psi34]))**2

    return psis

theta = np.linspace(0,np.pi,60)
phi = np.linspace(0,2*np.pi,60)
THETA, PHI = np.meshgrid(theta, phi)
